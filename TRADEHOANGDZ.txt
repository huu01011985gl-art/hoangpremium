local function AddDropdownOption(nameText)
	local OptionBtn = Instance.new("TextButton")
	OptionBtn.Parent = DropdownList
	OptionBtn.Size = UDim2.new(1, 0, 0, 28)
	OptionBtn.BackgroundTransparency = 1
	OptionBtn.Text = "  ".. nameText
	OptionBtn.TextColor3 = Color3.fromRGB(53, 33, 84)
	OptionBtn.Font = Enum.Font.Gotham
	OptionBtn.TextSize = 12
	OptionBtn.TextXAlignment = Enum.TextXAlignment.Left
	OptionBtn.TextTruncate = Enum.TextTruncate.AtEnd 
	OptionBtn.ZIndex = 51

	OptionBtn.MouseButton1Click:Connect(function()
		DropdownLabel.Text = "👤 Freeze Trade: ".. nameText
		DropdownList.Visible = false
		DropdownIcon.Text = "▼"
	end)
end

DropdownButton.MouseButton1Click:Connect(function()
	DropdownList.Visible = not DropdownList.Visible
	if DropdownList.Visible then
		DropdownIcon.Text = "▲"
		for _, child in pairs(DropdownList:GetChildren()) do
			if child:IsA("TextButton") then child:Destroy() end
		end
		
		AddDropdownOption("Unknown")
		for _, player in pairs(Players:GetPlayers()) do
			if player ~= LocalPlayer then
				AddDropdownOption(player.Name)
			end
		end
		DropdownList.CanvasSize = UDim2.new(0, 0, 0, #DropdownList:GetChildren() * 28)
	else
		DropdownIcon.Text = "▼"
	end
end)

local AddFruitBtn = Instance.new("TextButton")
AddFruitBtn.Size = UDim2.new(1, -30, 0, 34)
AddFruitBtn.Position = UDim2.new(0, 15, 0, 140)
AddFruitBtn.BackgroundColor3 = Color3.fromRGB(186, 224, 247) 
AddFruitBtn.Text = "🍎 Add Fruit"
AddFruitBtn.TextColor3 = Color3.fromRGB(53, 33, 84) 
AddFruitBtn.Font = Enum.Font.GothamBold
AddFruitBtn.TextSize = 13
AddFruitBtn.Parent = MainFrame
Instance.new("UICorner", AddFruitBtn).CornerRadius = UDim.new(0, 8)

local isCooldown = false
AddFruitBtn.MouseButton1Click:Connect(function()
	if isCooldown then return end
	isCooldown = true
	local duration = 5.0
	local startTime = tick()
	local originalColor = AddFruitBtn.BackgroundColor3
	
	while tick() - startTime < duration do
		local timeLeft = duration - (tick() - startTime)
		AddFruitBtn.Text = string.format("⏳ Adding Fruit... %.1fs", timeLeft)
		task.wait(0.03)
	end
	
	AddFruitBtn.Text = "✅ Added Fruit"
	AddFruitBtn.BackgroundColor3 = Color3.fromRGB(155, 246, 255) 
	task.wait(1)
	AddFruitBtn.Text = "🍎 Add Fruit"
	AddFruitBtn.BackgroundColor3 = originalColor
	isCooldown = false
end)
